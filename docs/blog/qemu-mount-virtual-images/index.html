Qemu/Mount Virtual Images

apt-get install -y libguestfs-tools

# Mounting
guestmount -a windows.qcow2 /mnt/

# Mounting read-only mode
guestmount -a windows.qcow2 /mnt/ --ro

# Unmount
guestunmount /mnt
